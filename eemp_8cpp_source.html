50

53 #endif

55  // compute domain range over all generators
56  unsigned dom_min = UINT_MAX;
57  for (PartialPerm const &gen : generators)
58  dom_min = std::min(dom_min, gen.dom_min());

60  // data structures for efficient component membership testing

64  struct ComponentElemId {
65  ComponentElemId(unsigned id, std::size_t hash) : id(id), hash(hash) {}
66 
67  unsigned id;
68  std::size_t hash;
69  };
70 
71  std::vector<std::vector<ComponentElemId>> elem_ids(dom_max + 1u);
72 
73  std::size_t hash = util::container_hash(alpha.begin(), alpha.end());
74  elem_ids[alpha.size()].push_back(ComponentElemId(0u, hash));
75 
76  // if beta is a component element, set 'id' to it's component index,
77  // otherwise add add it to the component
78  auto in_component = [&](std::vector<unsigned> const &beta, unsigned &id) {
79  bool contained = true;
80

85  contained = false;

88  auto it = std::find_if(
89  elem_ids[size_idx].begin(), elem_ids[size_idx].end(),
90  [&](ComponentElemId const &cei) { return cei.hash == beta_hash; });
91 
92  if (it == elem_ids[size_idx].end())
93  contained = false;
94  else
95  id = it->id;
96  }
97 
98  if (contained)
99  return true;
100 
101  DBG(TRACE) << "Adjoining " << beta;
102  elem_size_present[size_idx] = true;
103  elem_ids[size_idx].push_back(ComponentElemId(id, beta_hash));
104 
105  return false;
106  };
107 
108  // main loop
109  std::vector<std::vector<unsigned>> component {alpha};
110  decltype(schreier_tree.data) schreier_tree_data;
111  decltype(orbit_graph.data) orbit_graph_data(generators.size());
112 
113  unsigned i = 0u, n = 0u;
114  while (i < component.size()) {
115  std::vector<unsigned> beta = component[i];
116  DBG(TRACE) << "Looking at Component element " << beta
117  << " (i = " << i << ')';
118 
119  for (auto j = 0u; j < generators.size(); ++j) {
120  DBG(TRACE) << "Considering generator " << generators[j];
121 
122  auto beta_prime(generators[j].image<std::vector>(beta.begin(),
123  beta.end()));
124 
125  unsigned id = static_cast<unsigned>(component.size());
126 
127  if (!in_component(beta_prime, id)) {
128  component.push_back(beta_prime);
129 
130  ++n;
131 
132  schreier_tree_data.push_back(std::make_pair(i, j));
133  DBG(TRACE) << "v_" << n + 1u << " = " << i + 1u;
134  DBG(TRACE) << "w_" << n + 1u<< " = " << j + 1u;
135 
136  orbit_graph_data[j].push_back(n);
137  DBG(TRACE) << "g_" << i + 1u << "," << j + 1u << " = " << n + 1u;
138 
139  } else {
140  DBG(TRACE) << beta_prime << " already processed";
141 
142  orbit_graph_data[j].push_back(id);
143  DBG(TRACE) << "g_" << i + 1u << "," << j + 1u << " = " << id + 1u;
144  }
145  }
146 
147  ++i;
148  }
149 
150  schreier_tree.data = schreier_tree_data;
151 
152 #ifndef NDEBUG
153  DBG(TRACE) << "Resulting action component";
154  for (auto i = 0u; i < component.size(); ++i)
155  DBG(TRACE) << i + 1u << ": " << component[i];
156 #endif
157 
158  DBG(TRACE) << "Resulting schreier tree:\n" << schreier_tree;
159 
160  orbit_graph.data = orbit_graph_data;
161 
162  DBG(TRACE) << "Resulting orbit graph:\n" << orbit_graph;
163 
164  return component;
165 }

204

206  std::vector<unsigned> vertex_map(scc.size(), 0u);
207 
208  DBG(TRACE) << "Constructing temporary subgraph";
209

211  if (scc[j] == scc[i]) {
212  scc_i.push_back(j);
213

215  vertex_map[j] = v + 1u;
216

218  }
219  }
220

222

224  for (unsigned row = 0; row < orbit_graph.data.size(); ++row) {
225  unsigned dest = orbit_graph.data[row][source];
226 
227  unsigned source_vertex = vertex_map[source];
228  unsigned dest_vertex = vertex_map[dest];
229

231  boost::add_edge(dest_vertex - 1u, source_vertex - 1u, {row}, g);
232 
233  DBG(TRACE) << "Added edge "
234  << source_vertex - 1u << " => " << dest_vertex - 1u
235  << " (" << source + 1u << " => " << dest + 1u
236  << " (" << row + 1u << "))";
237  }
238  }
239  }
240

242  SchreierTree spanning_tree;

245

247 
248  boost::random_spanning_tree(
249  g, re, boost::root_vertex(vertex_map[i] - 1u).predecessor_map(&pred[0]));
250

252  << std::vector<unsigned>(pred.begin() + 1, pred.end());
253

255  spanning_tree.data =
256  std::vector<std::pair<unsigned, unsigned>>(scc.size() - 1u);
257 
258  for (auto j = 1u; j < pred.size(); ++j) {
259  auto child(boost::vertex(j, g));
260  auto parent(boost::vertex(pred[j], g));
261  auto edge(boost::edge(child, parent, g));
262 
263  assert(edge.second && "edge exists");
264 
265  spanning_tree.data[g[child].i - 1u] =
266  std::make_pair(g[parent].i, g[edge.first].gen);

276  std::vector<PartialPerm> const &generators, unsigned dom_max, unsigned target)
277 {

291 PermGroup schreier_generators(unsigned i,
292  std::vector<PartialPerm> const &generators, unsigned dom_max,
293  std::vector<std::vector<unsigned>> const &action_component,
294  SchreierTree const &schreier_tree, OrbitGraph const &orbit_graph,
295  std::vector<unsigned> const &sccs)
296 {

298

300 
301  if (im.empty()) {
302  DBG(TRACE) << "=> Returning empty permutation group";
303  return PermGroup();
304  }
305 
306  unsigned im_max = im.back();
307 
308  std::vector<unsigned> scc;
309  for (unsigned j = 0u; j < sccs.size(); ++j) {
310  if (sccs[j] == sccs[i])
311  scc.push_back(j);
312  }
313 
314 #ifndef NDEBUG
315  std::vector<std::vector<unsigned>> _scc(scc.size());
316  for (auto j = 0u; j < scc.size(); ++j)
317  _scc[j] = action_component[scc[j]];
318 
319  DBG(TRACE) << "Strongly connected component of x is: " << _scc;
320 #endif
321 
322  PermSet sg_gens;
323 
324  for (auto j = 0u; j < scc.size(); ++j) {
325  for (auto k = 0u; k < generators.size(); ++k) {
326 
327  unsigned l = orbit_graph.data[k][scc[j]];
328  if (sccs[l] != sccs[i])
329  continue;
330 
331  DBG(TRACE) << "Found generator for j/i(j)/k/l = "
332  << j + 1u << '/'
333  << scc[j] + 1u << '/'
334  << k + 1u << '/'
335  << l + 1u;
336 
337  PartialPerm ui(schreier_trace(
338  scc[j], schreier_tree, generators, dom_max, i));
339  PartialPerm ul(schreier_trace(
340  l, schreier_tree, generators, dom_max, i));
341 
342  DBG(TRACE) << "where:";
343  DBG(TRACE) << "u_i(j) = " << "u_" << scc[j] + 1u << " = " << ui;
344  DBG(TRACE) << "x_k = " << "x_" << k + 1u << " = " << generators[k];
345  DBG(TRACE) << "~u_l = " << "u_" << l + 1u << " = " << ~ul;
346 
347  PartialPerm sg(ui * generators[k] * ~ul);
348  sg = sg.restricted(im.begin(), im.end());
349 
350  DBG(TRACE) << "Schreier generator is: " << sg;
351 
352  if (!sg.id())
353  sg_gens.emplace(sg.to_perm(im_max));
354  }
355  }
356 
357  PermGroup res(im_max, sg_gens);
358 
359  DBG(TRACE) << "=> Returning:";
360  DBG(TRACE) << res;
361 
362  return res;
363 }
